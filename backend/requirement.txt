

//create a unit master table
CREATE TABLE units (

    id BIGSERIAL PRIMARY KEY,

    name VARCHAR(50) NOT NULL UNIQUE,

    symbol VARCHAR(20) NOT NULL UNIQUE,

    display_order INT DEFAULT 0,

    status INT DEFAULT 901,

    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,

    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP

);


//create a index of product add
CREATE INDEX idx_products_category
ON products(category_id);

CREATE INDEX idx_products_brand
ON products(brand_id);

CREATE INDEX idx_products_anime_series
ON products(anime_series_id);

CREATE INDEX idx_products_status
ON products(status);

CREATE INDEX idx_products_sku
ON products(sku);


//product create table
CREATE TABLE products (

    id BIGSERIAL PRIMARY KEY,

    name VARCHAR(255) NOT NULL,

    slug VARCHAR(255) NOT NULL UNIQUE,

    sku VARCHAR(100) NOT NULL UNIQUE,

    description TEXT,

    short_description VARCHAR(255),

    price NUMERIC(10,2) NOT NULL,

    discount_price NUMERIC(10,2),

    stock INT NOT NULL DEFAULT 0,

    size_value VARCHAR(20) NOT NULL,

    unit_id BIGINT NOT NULL,

    category_id BIGINT NOT NULL,

    brand_id BIGINT NOT NULL,

    anime_series_id BIGINT NOT NULL,

    weight DECIMAL(10,2),

    is_featured BOOLEAN DEFAULT FALSE,

    status INT NOT NULL DEFAULT 914,

    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,

    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,

    CONSTRAINT fk_product_unit
        FOREIGN KEY (unit_id)
        REFERENCES units(id)
        ON UPDATE CASCADE
        ON DELETE RESTRICT,

    CONSTRAINT fk_product_category
        FOREIGN KEY (category_id)
        REFERENCES categories(id)
        ON UPDATE CASCADE
        ON DELETE RESTRICT,

    CONSTRAINT fk_product_brand
        FOREIGN KEY (brand_id)
        REFERENCES brands(id)
        ON UPDATE CASCADE
        ON DELETE RESTRICT,

    CONSTRAINT fk_product_anime_series
        FOREIGN KEY (anime_series_id)
        REFERENCES anime_series(id)
        ON UPDATE CASCADE
        ON DELETE RESTRICT

);

//image product create
CREATE TABLE product_images (

    id BIGSERIAL PRIMARY KEY,

    product_id BIGINT NOT NULL,

    image_url TEXT NOT NULL,

    display_order INT DEFAULT 1,

    is_primary BOOLEAN DEFAULT FALSE,

    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,

    CONSTRAINT fk_product_image
        FOREIGN KEY (product_id)
        REFERENCES products(id)
        ON DELETE CASCADE

);
